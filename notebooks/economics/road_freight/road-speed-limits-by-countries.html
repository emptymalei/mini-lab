def clean_speed(speed, mode=None):
    if mode is None:
        mode = "max"
    
    if isinstance(speed, str):
        res = re.sub('\[[^>]+\]', '', speed)
        res = re.sub('\([^>]+\)', '', res)

        res = [int(e) for e in re.split("[^0-9]", res) if e != '']
    else:
        res = None
        
    if res:
        if mode == "max":
            res = max(res)
        elif mode == "min":
            res = min(res)
        else:
            raise NotImplementedError(f"mode = {mode} has not yet been implemented")
    else:
        res = None
        
    return res
